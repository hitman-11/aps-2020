#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int arr[n];
        
        for(int i=0;i<n;i++)
        {
            
            cin>>arr[i];
        }
        
        vector<int>arr1;
        for(int i=0;i<n;i++)
        {
            if(arr[i]==1)
            {
                arr1.push_back(i);
            }
        }
       /* for(int i=0;i<n;i++)
        {
            cout<<arr1[i];
        }*/
        int flag=0;
        for(int i=0;i<arr1.size()-1;i++)
        {
            if((arr1[i+1]-arr1[i])<6)
            {
                flag = 1;
                break;
            }
            else
            {
                flag = 0;
                
            }
            
        }
        
        if(flag==1)
        {
            cout<<"NO"<<endl;
        }
        else
        {
            cout<<"YES"<<endl;
        }
    }
    return 0;
}
